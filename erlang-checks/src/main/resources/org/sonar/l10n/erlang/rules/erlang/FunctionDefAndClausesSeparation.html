Try to separate your code blocks for readability.

  • Always use 1 blank line to separate function definitions
  • Do not use blank lines to separate function clauses

Like: 

-module(a).
-export([int/0]).

-spec int(value()) -> ConvertedValue::term().
int(Var) -> to_type(int, Var).

-spec hello() -> atom().
hello([])->
  Code;
hello(A) ->
  Code.
hello(a,b) -> %issue, no blank line before declaration
  Code;
hello(c, _) ->
  Code.
bello([])-> %issue, no blank line before declaration
  code;
bello(A) ->
  {A, code}.

%comment is
%not an issue
hello2(a)->
  Code;

hello2(_) -> %issue, blank line before function clause
  Code.


bello2([])-> %issue, two blank lines before declaration
  code;
%not an issue
bello2(A) ->
  {A, code}.
